    const stages = [
      'pending_head',
      'pending_admin',
      'pending_comptroller',
      'pending_hr',
      'pending_exec',
      'approved'
    ];
    
    let currentIndex = stages.indexOf(currentStage);
    let nextIndex = currentIndex + 1;
    
    while (nextIndex < stages.length) {
      const nextStage = stages[nextIndex];
      
      // Check if this stage already has a signature
      if (hasExistingSignature(request, nextStage)) {
        console.log(`Skipping ${nextStage} - signature exists`);
        nextIndex++;
        continue;
      }
      
      return nextStage;
    }
    
    return 'approved';
  }
  
  function hasExistingSignature(request, stage) {
    switch(stage) {
      case 'pending_head':
        return request.head_signature !== null;
      case 'pending_comptroller':
        return request.comptroller_signature !== null;
      case 'pending_hr':
        return request.hr_signature !== null;
      case 'pending_exec':
        return request.exec_signature !== null;
      default:
        return false;
    }
  }

      req.position_title as requester_position,
      
      head.name as head_name,
      head.profile_picture as head_photo,
      head.position_title as head_position,
      r.head_signature,
      
      comp.name as comptroller_name,
      comp.profile_picture as comptroller_photo,
      comp.position_title as comptroller_position,
      r.comptroller_signature,
      
      hr.name as hr_name,
      hr.profile_picture as hr_photo,
      hr.position_title as hr_position,
      r.hr_signature,
      
      exec.name as exec_name,
      exec.profile_picture as exec_photo,
      exec.position_title as exec_position,
      r.exec_signature
      
    FROM requests r
    LEFT JOIN users req ON r.requester_id = req.id
    LEFT JOIN users head ON r.head_approved_by = head.id
    LEFT JOIN users comp ON r.comptroller_approved_by = comp.id
    LEFT JOIN users hr ON r.hr_approved_by = hr.id
    LEFT JOIN users exec ON r.exec_approved_by = exec.id
    WHERE r.id = $1;
